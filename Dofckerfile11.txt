# Stage 1: Build environment for development
FROM maven:3.8.4-openjdk-11 AS dev_build

WORKDIR /app

# Copy only the pom.xml to cache dependencies
COPY pom.xml .

# Fetch dependencies
RUN mvn dependency:go-offline

# Copy the source code and build application
COPY src ./src
RUN mvn package -DskipTests

# Stage 2: Build environment for production
FROM maven:3.8.4-openjdk-11 AS prod_build

WORKDIR /app

# Copy only the pom.xml to cache dependencies
COPY pom.xml .

# Fetch dependencies
RUN mvn dependency:go-offline

# Copy the source code and build optimized production JAR
COPY src ./src
RUN mvn package -DskipTests -Pproduction

# Stage 3: Runtime environment
FROM openjdk:11-jre-slim

WORKDIR /app

# Create a non-root user
RUN adduser --system --group javauser
USER javauser

# Copy the optimized production JAR from production build stage
COPY --from=prod_build /app/target/*.jar app.jar

# Expose application port
EXPOSE 8080

# Command to run the application
CMD ["java", "-jar", "app.jar"]
